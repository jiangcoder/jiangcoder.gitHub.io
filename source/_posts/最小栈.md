## 最小栈
```java
class MinStack {

    /** initialize your data structure here. */
    
    private Node node;
    public MinStack() {
        
    }
    
    public void push(int x) {
        if (node == null)
            node = new Node(x, x);
        else node = new Node(x, Math.min(x, node.min), node);
    }
    
    public void pop() {
        node = node.next;
    }
    
    public int top() {
        return node.val;
    }
    
    public int getMin() {
        return node.min;
    }

    private class Node {
        int val;
        int min;
        Node next;
        private Node(int val, int min) {
            this(val, min, null);
        }
        private Node(int val, int min, Node next) {
           this.val = val;
           this.min = min;
           this.next = next;
        }
    }
}
```
